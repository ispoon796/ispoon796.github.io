算法模板-3

2022/9/20 c/c++ codeforce algorithm

主要介绍了图论中比较常见的一些算法，包括图的遍历、最短路算法、最小生成树算法、二分图算法。

图论

根据解决问题的不同，我们可以将算法分为如下的几类，以及其对应的可以解决的问题。

图的建立

在介绍常见的算法之前，我们需要首先介绍几种常见的建图方式
· 邻接矩阵
· 邻接表
· 结构体

邻接矩阵

对于邻接矩阵而言，其主要针对的是稠密图而言，其中g[i][j]=c表示的是节点 i 和节点 j 之间存在一条两边，且对应的边的长度为c。
对于无向图而言，一般而言，我们可以需要同时赋值 g[i][j] 和 g[j][i] 。为了避免在输入的过程中出现重边的情况，对应的边值初始化的方法是：
g[i][j]=g[j][i]=min(g[i][j],c);
对于有向图而言，我们只需要赋值g[i][j]即可，同时，我们也需要考虑可能存在的重边情况。
g[i][j]=min(g[i][j],c);

邻接表

对于邻接表而言，我们主要采用的数组模拟的方式实现，即通过数组的方式模拟链表，从而构建对应的图结构。对应的邻接表的初始化代码为：

const int N=2e5+10;
int h[N],e[N],ne[N],w[N],idx;
void init(){
    memset(h,-1,sizeof h);
}

其中h[a]表示的是以a为起点的链表的起始位置，e表示的当前链表所指向的值，ne表示的当前链表对应的下一个链表的值，w表示的是对应的权重，idx表示的开辟的可能空间数，其对应的数据结果的具体表现形式如下所示：

通过这种方式，我们可以较好地存储对应的稀疏图结构。
特别的，对于该图而言，我们对应的添加方式为：

void add(int a,int b,int c){
    e[idx]=b,w[idx]=c,ne[idx]=h[a],h[a]=idx++;
}

结构体

对于结构体而言，其主要针对的是不太注重图的物理结构的算法，（比如Bellman_ford算法以及Kruskal算法等等），其对应的存储形式如下所示：

struct Edge{
    int a,b,w;
};

对于某些不需要考虑点之间联通关系的算法，该算法可以较快的进行存储。

图的遍历

介绍完对应的图的存储方式之后，我们介绍基于邻接表的方式下，对应的图的遍历过程。其他的遍历方式类似，这里就不多做展示了。

邻接表

对于邻接表而言，对应的bfs的遍历方式为：

BFS

#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
#include<queue>
using namespace std;
const int N=2e5+10;
int h[N],e[N],ne[N],idx;
void add(int a,int b){
    e[idx]=b,ne[idx]=h[a],h[a]=idx++;
}
int n,m;
bool st[N];
int bfs(int a){
    queue<int>que;
    que.push(a);
    // 通过这样的方式实现遍历的结果
    while(que.size()){
        int aa=que.front();que.pop();
        for(int i=h[aa];i!=-1;i=ne[i]){
            int j=e[i];
            if(!st[j]) {
                que.push(j);
                st[j]=true;
            }
        }
    }
}
int main(){
    memset(h,-1,sizeof h);
    scanf("%d %d",&n,&m);
    for(int i=0;i<m;i++){
        int a,b;
        scanf("%d%d",&a,&b);
        add(a,b);
    }
    return 0;
}

DFS

对于DFS而言，对应的遍历方式如下所示：

#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
#include<queue>
using namespace std;
const int N=2e5+10;
int h[N],e[N],ne[N],idx;
void add(int a,int b){
    e[idx]=b,ne[idx]=h[a],h[a]=idx++;
}
int n,m;
int st[N];
void dfs(int a){
    for(int i=h[a];i!=-1;i=ne[i]){
        int j=e[i];
        if(!st[j]){
            st[j]=true;
            dfs(j);// 通过这样的方式实现遍历的操作
        }
    }
}
int main(){
    scanf("%d%d",&n,&m);
    memset(h,-1,sizeof h);
    while(m--){
        int a,b;
        scanf("%d%d",&a,&b);
        add(a,b);
    }
    return 0;
}

最短路问题

单源最短路问题

对于单源最短路问题而言，我们主要需要考虑其中是否会出现负权的边。因而其对应的

dijkstra算法的实现

对于最短路问题而言，我们可以根据起点的个数分为单源最短路问题，其中对于单源最短路而言，其可以根据对应的边权的种类分成全为正值的最短路以及存在负权的最短路问题。对于全为正值的最短路问题而言，我们可以采用dijsktra算法解决，根据存储图的结构，我们可以将算法分类为

· 朴素的单源最短路问题 –> 对于稠密图而言
· 基于堆优化的最短路问题 –> 对于稀疏图而言

对于朴素的单源最短路问题而言，其对应的解可以表示为：

#include<cstring>
#include<algorithm>
#include<cstdio>
#include<queue>
using namespace std;
const int N=2e3+10;
int g[N][N];
int dist[N];
bool st[N];
int n,m;
int dijkstra(){
    memset(dist,0x3f,sizeof dist);
    dist[1]=0;
    for(int i=0;i<n;i++){
        int t=-1;
        for(int j=1;j<=n;j++){
            if(!st[j]&&(t==-1||dist[t]>dist[j]))
                t=j;
        }
        st[t]=true;
        for(int j=1;j<=n;j++){
            dist[j]=min(dist[j],dist[t]+g[t][j]);
        }
    }
    return dist[n];
}
int main(){
    scanf("%d%d",&n,&m);
    memset(g,0x3f,sizeof g);
    for(int i=0;i<m;i++){
        int a,b,c;
        scanf("%d%d%d",&a,&b,&c);
        g[a][b]=min(g[a][b],c);
    }
    int t=dijkstra();
    if(t==0x3f3f3f3f) printf("-1\n");
    else printf("%d\n",t);
    return 0;
}

对于节点数相对较多的情况，此时如果还是采用邻接矩阵的方式，容易出现存储错误，所以，我们采用的是邻接表的方式实现（借助于堆排序的方式）。需要注意的是，其主要针对的是稠密图的形式。

// 采用的是邻接矩阵的方式进行存储
#include<cstring>
#include<algorithm>
#include<cstdio>
#include<queue>
using namespace std;
const int N=2e5+10;
int h[N],e[N],ne[N],w[N],idx;
int dist[N];
bool st[N];
typedef pair<int,int>PII;

void add(int a,int b,int c){
    e[idx]=b,ne[idx]=h[a],w[idx]=c,h[a]=idx++;
}
int n,m;
int dijkstra(){
    memset(dist,0x3f,sizeof dist);
    dist[1]=0;
    priority_queue<PII,vector<PII>,greater<PII>>que;
    que.push({0,1});
    while(que.size()){
        auto aa=que.front();que.pop();
        int ver=aa.second,distance=aa.first;
        if(st[ver]) continue;
        st[ver]=true;
        for(int i=h[ver];i!=-1;i=ne[i]){
            int j=e[i];
            if(dist[j]>distance+w[i]){
                dist[j]=distance+w[i];
                que.push({dist[j],j});
            }
        }
    }
    return dist[n];
}
int main(){
    memset(h,-1,sizeof h);
    scanf("%d%d",&n,&m);
    for(int i=0;i<m;i++){
        int a,b,c;
        scanf("%d%d%d",&a,&b,&c);
        add(a,b,c);
    }
    int t=dijkstra();
    if(t==0x3f3f3f3f) printf("-1\n");
    else printf("%d\n",t);
}

负权图的实现方式

如果图中存在负权图的时候，此时就不能通过dijkstra算法去进行实现。其常用的算法主要有bellenman_ford以及spfa算法。
其中bellenman_ford算法采用的是结构体的方式实现对图表的存储，其对应的实现方式为：
bellenman_ford 算法模板

#include<cstring>
#include<algorithm>
#include<cstdio>
#include<queue>
using namespace std;
const int N=2e5+10;
struct Edge{
    int a,b,c;
}e[N];
int n,m,k;
int dist[N],backup[N];
int bellenman_ford(){
    memset(dist,0x3f,sizeof dist);
    dist[1]=0; // 一定要进行初始化
    for(int i=0;i<k;i++){
        memcpy(backup,dist,sizeof dist);
        for(int j=0;j<m;j++){
            int aa=e[j].a,bb=e[j].b,cc=e[j].c;
            dist[bb]=min(dist[bb],backup[aa]+cc);
        }
    }
    return dist[n];
}
int main(){
    scanf("%d%d%d",&n,&m,&k);
    for(int i=0;i<m;i++){
        int a,b,c;
        scanf("%d%d%d",&a,&b,&c);
        e[i]={a,b,c};
    }
    int t=bellenman_ford();
    if(t>0x3f3f3f3f/2) printf("impossible");
    else printf("%d\n",t);
    return 0;
}

同样的，我们也可以通过spfa算法实现对其的快速搜索：

#include<cstring>
#include<algorithm>
#include<cstdio>
#include<queue>
using namespace std;
const int N=2e5+10;
int h[N],e[N],w[N],ne[N],idx;
int dist[N];
bool st[N];
int n,m;
// 添加相应的结果
void add(int a,int b,int c){
    e[idx]=b,ne[idx]=h[a],w[idx]=c,h[a]=idx++;
}

int spfa(){
    memset(dist,0x3f,sizeof dist);
    dist[1]=0;
    queue<int>q;
    q.push(1);
    st[1]=true;// 只有被修改过后才能放入
    while(q.size()){
        int aa=q.front();q.pop();
        st[aa]=false;
        for(int i=h[aa];i!=-1;i=ne[i]){
            int j=e[i];
            if(dist[j]>dist[aa]+w[i]){
                dist[j]=dist[aa]+w[i];
                if(!st[j]){
                    q.push(j);
                    st[j]=true;
                }
            }
        }    
    }
    return dist[n];
}
int main(){
    scanf("%d%d",&n,&m);
    memset(h,-1,sizeof h);
    for(int i=0;i<m;i++){
        int a,b,c;
        scanf("%d%d%d",&a,&b,&c);
        add(a,b,c);
    }
    int t=spfa();
    if(t>0x3f3f3f3f/2) puts("impossible");
    else printf("%d\n",t);
    return 0;
}

同时，spfa算法还可以判断当前循环中是否存在负环。其本质上就是需要对应的每个点的度数不得大于等于n

#include<cstring>
#include<algorithm>
#include<cstdio>
#include<queue>
using namespace std;
const int N=2e5+10;
int h[N],e[N],w[N],ne[N],idx;
int dist[N];
int cnt[N];
bool st[N];// 判断当前的状态
int n,m;
void add(int a,int b,int c){
    e[idx]=b,ne[idx]=h[a],w[idx]=c,h[a]=idx++;
}

bool spfa(){
    memset(dist,0x3f,sizeof dist);
    dist[1]=0;
    queue<int>q;
    for(int i=1;i<=n;i++){
        q.push(i);
        st[i]=true;
    }
    while(q.size()){
        int t=q.front();q.pop();
        st[t]=false;
        for(int i=h[t];i!=-1;i=ne[i]){
            int j=e[i];
            if(dist[j]>dist[t]+w[i]){
                dist[j]=dist[t]+w[i];
                if(!st[j]){
                    q.push(j);
                    st[j]=true;
                    cnt[j]=cnt[t]+1;
                    if(cnt[j]>=n) return true; // 如果是这样的话 就存在负环
                }
            }
        }
    }
    return false;// 反之就不存在负环
}
int main(){
    scanf("%d%d",&n,&m);
    memset(h,-1,sizeof h);
    for(int i=0;i<m;i++){
        int a,b,c;
        scanf("%d%d%d",&a,&b,&c);
        add(a,b,c);
    }
    if(spfa()) puts("Yes");
    else puts("No");
    return 0;
}

多源最短路问题

对于多源最短路而言，我们主要采用的是floyd的方式实现，其对应的存储方式是采用的是邻接矩阵的方式。

#include<cstring>
#include<algorithm>
#include<cstdio>
#include<queue>
using namespace std;
const int N=2e3+10;
int g[N][N];
int n,m,k;
void floyd(){
    for(int k=1;k<=n;k++){
        for(int i=1;i<=n;i++){
            for(int j=1;j<=n;j++){
                g[i][j]=min(g[i][k]+g[k][j],g[i][j]);
            }
        }
    }
}
int main(){
    memset(g,0x3f,sizeof g);
    scanf("%d%d%d",&n,&m,&k);
    for(int i=1;i<=n;i++) g[i][i]=0;
    for(int i=0;i<m;i++){
        int a,b,c;
        scanf("%d%d%d",&a,&b,&c);
        g[a][b]=min(g[a][b],c);
    }
    floyd();
    while(k--){
        int a,b;
        scanf("%d%d",&a,&b);
        if(g[a][b]>0x3f3f3f3f/2) puts("impossible");
        else
            printf("%d\n",g[a][b]);
    }
    return 0;
}

最小生成树算法

对于最小生成树算法而言，一般而言，我们可以采用的算法主要包括prim算法以及kruskal算法。其中，考虑到对应的算法实现的复杂程度，对于prim算法而言，我们一般选择邻接矩阵的方式进行存储；对于kruskal算法而言，我们一般采用邻接表的方式存储。如下是对应部分的代码：

Prim算法

#include<iostream>
#include<cstring>
#include<algorithm>
#include<cstdio>
#include<queue>
using namespace std;
const int N=5e2+50;
const int INF=0x3f3f3f3f;
int g[N][N];
int n,m;
int dist[N];
int st[N];
// 对于所有的边进行处理
int prim(){
    memset(dist,0x3f,sizeof dist);
    int res=0;
    for(int i=0;i<n;i++){
        int t=-1;
        for(int j=1;j<=n;j++){
            if(!st[j]&&(t==-1||st[t]>st[j]))
                t=j;
        }
        if(i&&dist[t]==INF) return dist[t];
        if(i) res+=dist[t];
        st[t]=true;
        for(int j=1;j<=n;j++){
            dist[j]=min(dist[j],g[t][i]);
        }
    }
    return res;
}
int main(){
    scanf("%d%d",&n,&m);
    memset(g,0x3f,sizeof g);
    for(int i=0;i<m;i++){
        int a,b,c;
        scanf("%d%d%d",&a,&b,&c);
        g[a][b]=g[b][a]=min(g[a][b],c);
    }
    int t=prim();
    if(t==INF) printf("impossible\n");
    else printf("%d\n",t);
    return 0;
}

kruskal算法

// 需要注意的是结构体的边本质上是双线的，所以这里不需要进行额外的处理
#include<iostream>
#include<cstring>
#include<algorithm>
#include<cstdio>
#include<queue>
using namespace std;
const int N=2e5+10;

struct Edge{
    int a,b,c;
    bool operator<(const Edge E){
        return c<E.c;
    }
}e[N];
int n,m;
int f[x];
int find(int x){
    if(f[x]!=x) f[x]=findf(f[x]);
    return f[x];
}

int main(){
    scanf("%d%d",&n,&m);
    for(int i=0;i<m;i++){
        scanf("%d%d%d",&e[i].a,&e[i].b,&e[i].c);
    }
    sort(e,e+m);
    for(int i=1;i<=n;i++) f[i]=i;
    int res=0;
    int cnt=0;
    for(int i=0;i<m;i++){
        int a=e[i].a,b=e[i].b,c=e[i].c;
        int aa=find(a);
        int bb=find(b);
        if(aa!=bb){
            f[aa]=bb;
            res+=c;
            cnt++;
        }
        if(cnt==n-1){
            break;
        }
    }
    if(cnt==n-1) printf("%d\n",res);
    else printf("impossible\n");
    return 0;
}

二分图问题

判断一个图是不是二分图，我们常用的方法是染色法，因为对于二分图而言，其不存在对应的奇数环，所以我们可以根据这一个性质实现对于其的判断。

判断一个图是否为二分图

#include<iostream>
#include<cstring>
#include<algorithm>
#include<cstdio>
#include<queue>
using namespace std;
const int N=2e5+10;
int h[N],e[N],ne[N],idx;
int color[N];
void add(int a,int b){
    e[idx]=b,ne[idx]=h[a],h[a]=idx++;
}
int n,m;
int dfs(int t,int c){
    color[t]=c;// 初始染色
    for(int i=h[t];i!=-1;i=ne[i]){
        int j=e[i];
        if(!color[j]){
            if(!dfs(j,3-c)) return 0;
        }
        else if(color[j]==c) return 0;
    }
    return 1;
}
int main(){
    scanf("%d%d",&n,&m);
    memset(h,-1,sizeof h);
    for(int i=0;i<m;i++){
        int a,b;
        scanf("%d%d",&a,&b);
        add(a,b);
        add(b,a);
    }
    int flag=1;
    for(int i=1;i<=n;i++){
        if(!color[i]){
            if(!dfs(i,1)){
                flag=0;
                break;
            }
        }
    }
    if(flag) printf("Yes\n");
    else printf("No\n");
}

二分图最大匹配问题

#include<iostream>
#include<cstring>
#include<algorithm>
#include<cstdio>
#include<queue>
using namespace std;
const int N=2e5+10;
int match[N],st[N];
int h[N],e[N],ne[N],idx;
void add(int a,int b){
    e[idx]=b,ne[idx]=h[a],h[a]=idx++;
}
int n,m,k;
bool find(int x){
    for(int i=h[x];i!=-1;i=ne[i]){
        int j=e[i];
        if(!st[j]){
            st[j]=true;
            if(match[j]==0||find(match[j])) {
                match[j]=x;
                return true;
            }
        }
    }
    return false;
}
int main(){
    scanf("%d%d%d",&n,&m,&k);
    memset(h,-1,sizeof h);
    for(int i=0;i<k;i++){
        int a,b;
        scanf("%d%d",&a,&b);
        add(a,b);
    }
    int res=0;
    for(int i=1;i<=n;i++){
        memset(st,0,sizeof st);
        if(find(i)) res++;
    }
    printf("%d\n",res);
    return 0;
}
