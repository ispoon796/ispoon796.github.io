排序算法，二分算法以及高精度算法的模板实现

基础算法

主要实现了排序算法，二分算法以及相应的高精度计算

快速排序

函数实现

// 快速排序 平均时间复杂度O(nlogn)
void qsort(int a[],int l,int r){
    if(l>=r) return;
    int x=a[l],i=l-1,j=r+1;
    while(i<j){
        while(a[++i]<x);
        while(a[--j]<x);
        if(i<j) swap(a[i],a[j]);
    }
    qsort(a,l,j);
    qsort(a,j+1,r);
}

函数调用

qsort(a,0,n-1);

归并排序

函数实现

int tmp[N];
// 归并排序 平均时间复杂度O(nlogn)
void msort(int a[],int l,int r){
    if(l>=r) return;
    int mid=l+r>>1,i=l,j=mid+1,k=0;
    while(i<=mid&&j<=r){
        if(a[i]<a[j]) tmp[k++]=a[i++];
        else tmp[k++]=a[j++];
    }
    while(i<=mid) tmp[k++]=a[i++];
    while(j<=r) tmp[k++]=a[j++];
    for(int i=0,j=l;i<k;i++,j++){
        a[j]=tmp[i];
    }
}

函数调用

msort(a,0,n-1);

二分法

二分法本质上是需要找到对应的边界存在的位置，所以理论上二分法并不需要要求区间是单调的。
根据分类数据的不同，我们将问题分为整数二分和浮点数二分，对于整数二分而言，考虑到其存在的可行边界，所以整数二分存在两种模板。

· 找到右区间的边界点

int l=0,r=n-1;
while(l<r){
    int mid=l+r>>1;
    if(check(x)) r=mid;
    else l=mid+1;
}

· 找到左区间的边界点

int l=0,r=n-1;
while(l<r){
    int mid=l+r+1>>1;
    if(check(x)) l=mid;
    else l=mid-1;
}

其中+1的含义就是为了避免出现越界的情况。

高精度算法

高精度算法主要是为了解决正常求解下可能存在的溢出的运算，如下是对应加减乘除的实现过程。

高精度加法

函数实现

vector<int> add(vector<int>&a,vector<int>&b){
    vector<int> c;
    int t=0;
    for(int i=0;i<a.size()||i<b.size();i++){
        if(i<a.size()) t+=a[i];
        if(i<b.size()) t+=b[i];
        c.push_back(t%10);
        t/=10;
    }
    if(t) c.push_back(1);
    return c;
}

调用接口

for(int i=b.size()-1;i>=0;i--) B.push_back(b[i]-'0');
auto c=add(A,B);
for(int i=c.size()-1;i>=0;i--) printf("%d",c[i]);

函数实现

// 比较函数的大小
bool cmp(vector<int>&a,vector<int>&b){
    if(a.size()!=b.size()) return a.size()>b.size();
    for(int i=a.size()-1;i>=0;i--){
        if(a[i]!=b[i]) return a[i]>b[i];
    }
}
// 计算减法
vector<int> mul(vector<int> &a,vector<int>&b){
    vector<int>c;
    int t=0;
    for(int i=0;i<a.size();i++){
        t=a[i]-t;
        if(b.size()) t-=b[i];
        c.push_back((t+10)%10);
        if(t<0) t=1;
        else t=0;
    }
    while(c.size()>1&&c.back()==0) c.pop_back();
    return c;
}

高精度除法

函数实现

## 其中b表示除数，t表示余数
vector<int> div(vector<int>&a,int b,int &t){
    vector<int>c;
    int t=0;
    for(int i=a.size()-1;i>=0;i--){
        t=t*10+a[i];
        c.push_back(t/b);
        t%=b;
    }
    reverse(c.begin(),c.end());
    while(c.size()>1&&c.back()==0) c.pop_back();
    return c;
}
