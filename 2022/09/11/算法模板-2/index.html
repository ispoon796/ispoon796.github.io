前缀和

前缀和是通过预处理的方式，求解前缀和的数组的和，通过这样的方式，我们就可以在O(1)的复杂度的情况下求解出某段区间的的值。以一维前缀和为例：

一维前缀和


实现方式

for(int i=1;i<=n;i++){
    scanf("%d"，&a[i]);
    s[i]=s[i-1]+a[i];
}

调用方式

// 求解[l,r]区间和
printf("%d\n",s[r]-s[r-1]);

二维前缀和

实现方式

\\初始化方式
for(int i=1;i<=n;i++){
    for(int j=1;j<=m;j++){
        scanf("%d",&a[i][j]);
        s[i][j]+=s[i-1][j]+s[i][j-1]-s[i-1][j-1];
    }
}

调用方式

while(q--){
    int x1,y1,x2,y2;
    scanf("%d%d%d%d",&x1,&y1,&x2,&y2);
    printf("%d\n",s[x2][y2]+s[x1][y1]-s[x1][y2]-s[x2][y1]);
}

差分

差分是通过预处理的方式，可以在O(1)的复杂度上实现对区间上每个数进行修改。

一维差分

// 表示对于区间l，r上的所有数字进行操作，使得起都增加a
void insert(int l,int r,int a){
    s[l]+=a;
    s[r+1]-=a;
}

调用方式

\\初始化方式
for(int i=1;i<=r;i++){
    scanf("%d",&a[i]);
    insert(i,i,a[i]);
}
\\ 修改方式
while(q--){
    int l,r,c;
    scanf("%d%d%d",&l,&r,&c);
    insert(l,r,c);
}

二维差分

实现方式

void insert(int x1,int y1,int x2,int y2){
    b[x1][y1]+=c;
    b[x1][y2+1]-=c;
    b[x2+1][y1]-=c;
    b[x2+1][y2+1]+=c;
}

调用方式

\\初始化方式
for(int i=1;i<=n;i++){
    for(int j=1;j<=m;j++){
        scanf("%d",&a[i][j]);
        insert(i,j,i,j,a[i][j]);
    }
}
\\修改方式
while(q--){
    int x1,y1,x2,y2,c;
    scanf("%d%d%d%d",&x1,&y1,&x2,&y2,&c);
    insert(x1,y1,x2,y2,c);
}
